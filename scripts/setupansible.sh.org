#!/bin/bash

function installSoftwares() {
  apt-get -y update
  apt-get -y install software-properties-common python-software-properties
  add-apt-repository ppa:ansible/ansible -y
  apt-get -y update
  apt-get install -y ansible git python-pip python2.7-dev libssl-dev vim telnet net-tools
}

installSoftwares
